SpingBoot 搭建 Websocket ​

引入 pom

xml
<dependency>
    <groupId>org.springframework.boot</groupId>
    <artifactId>spring-boot-starter-websocket</artifactId>
</dependency>

创建 service 端

java
/**
 * @author shunzi
 * @date 2023/11/24 10:36
 */
@Component
@Slf4j
@ServerEndpoint("/websocket/test/")
// @ServerEndpoint("/websocket/test/{userId}")
public class MyController {

    /**
     * userId 和 session
     */
    private static final Map<String, Session> sessionMap = new ConcurrentHashMap<>();

    /**
     * session.getId 和 userId
     */
    private static final Map<String, String> sessionIdMap = new ConcurrentHashMap<>();


    /**
     * 链接成功调用的方法
     *
     * @PathParam(value = "userId") String userId 参数
     */
    @OnOpen
    public void onOpen(Session session) {
        try {
            // 获取参数
            final String queryString = session.getQueryString();
            // 解析参数
            final Map<String, String> stringStringMap = this.explainQueryStringFromSession(queryString);
            final String userid = stringStringMap.get("userId");
            if (userid != null) {
                sessionMap.put(userid, session);
                sessionIdMap.put(session.getId(), userid);
            } else {
                session.getAsyncRemote().sendText("参数错误");
                session.close();
            }
            log.info("【websocket消息】有新的连接, userId:{}, queryString:{}, sessionMap:{}, this:{}", userid, session.getQueryString(), sessionMap, this);
            this.sendAsyncMessageByAll("wahaha");
        } catch (Exception e) {
            log.error("onOpen:{}", e);
        }
    }

    /**
     * 解析参数
     *
     * @param str
     * @return
     */
    private Map<String, String> explainQueryStringFromSession(String str) {
        if (!StringUtils.hasText(str)) {
            return Collections.emptyMap();
        }
        Map<String, String> paramMap = new HashMap<>();
        final String[] paramsSplit = str.split("&");
        for (String params : paramsSplit) {
            final String[] param = params.split("=");
            if (param.length == 2) {
                paramMap.put(param[0], param[1]);
            }
        }
        return paramMap;
    }


    /**
     * 链接关闭调用的方法
     */
    @OnClose
    public void onClose(Session session) {
        try {
            final String remove = sessionIdMap.remove(session.getId());
            final Session remove1 = sessionMap.remove(remove);
            log.info("【websocket消息】连接断开 sessionId:{}, remove1:{}", session.getId(), remove1);
        } catch (Exception e) {
            log.error("onClose:{}", e);
        }
    }

    /**
     * 收到客户端消息后调用的方法
     *
     * @param message
     * @param
     */
    @OnMessage
    public void onMessage(String message, Session session) {
        log.info("【websocket消息】收到客户端消息:{}, sessionId:{}, queryString:{}", message, session.getId(), session.getQueryString());
        try {
            session.getBasicRemote().sendText(message);
        } catch (Exception e) {
            log.error("onMessage:{}", e);
        }
    }

    /**
     * 发送错误时的处理
     *
     * @param session
     * @param error
     */
    @OnError
    public void onError(Session session, Throwable error) {
        log.error("用户错误,原因:{}", error);
    }


    /**
     * 发送异步消息
     */
    public void sendAsyncMessageByAll(String message) {
        for (Map.Entry<String, Session> stringSessionEntry : sessionMap.entrySet()) {
            stringSessionEntry.getValue().getAsyncRemote().sendText(message);
        }
    }

    /**
     * 发送同步消息
     */
    private void sendBasicMessageByAll(String message) {
        try {
            for (Map.Entry<String, Session> stringSessionEntry : sessionMap.entrySet()) {
                stringSessionEntry.getValue().getBasicRemote().sendText(message);
            }
        } catch (Exception e) {
            log.error("sendBasicMessageByAll:{}", e);
        }
    }
}
