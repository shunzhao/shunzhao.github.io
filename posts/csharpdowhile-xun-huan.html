C# do while 循环 ​

记一个C#商城购买的例子 用do while 循环 刚开始自己写 就是不知道如何下手，思路没跟上， 应该没静下心来，后来看了别人写的思路后 觉得好简单 于是自己动手写了一个，记下来 以后不懂再看看

和他不同的是：

他的判断如果余额小于购买的商品金额提示购买失败后用了一个continue来跳过这个循环到下一个循环，这点我理解不了，不知道正确性，特此记录。

c#
        //给予一个初始金额
        int rmb = 1000;
        //开始循环
        do
        {
            //商品列表
            Console.WriteLine("欢迎来到商城");
            Console.WriteLine("请输入序号进行购买");
            Console.WriteLine("1.可乐 100");
            Console.WriteLine("2.雪碧 200");
            Console.WriteLine("3.芬达 300");
            Console.WriteLine("4.牛奶 400");
            //接收输入的序号
            string jq = Console.ReadLine();
            //多分支选择
            switch (jq)
            {
                case "1":
                    //判断余额是否小于购买的商品金额
                    if (rmb<100)
                    {
                        Console.WriteLine("购买失败");
                    }
                    else
                    {
                        //购买成功后减去商品金额后再对rmb赋值
                        rmb = rmb - 100;
                        Console.WriteLine("购买成功，你还剩下{0}rmb",rmb);
                    }
                    break;
                case "2":
                    if (rmb < 200)
                    {
                        Console.WriteLine("购买失败");
                    }
                    else
                    {
                        rmb = rmb - 200;
                        Console.WriteLine("购买成功，你还剩下{0}rmb", rmb);                            
                    }
                    break;
                case "3":
                    if (rmb < 300)
                    {
                        Console.WriteLine("购买失败");
                    }
                    else
                    {
                        rmb = rmb - 300;
                        Console.WriteLine("购买成功，你还剩下{0}rmb", rmb);
                    }
                    break;
                case "4":
                    if (rmb < 400)
                    {
                        Console.WriteLine("购买失败");
                    }
                    else
                    {
                        rmb = rmb - 400;
                        Console.WriteLine("购买成功，你还剩下{0}rmb", rmb);
                    }
                    break;
                default:
                    Console.WriteLine("搞事情");
                    break;
            }
        } while (rmb>0);
        Console.WriteLine("余额不足无法继续购买");
